    } else if (target === 'vllm') {
      msg = 'Stopping Ollama Engine & starting vLLM Core...';
    } else if (target === 'stop_all') {
      msg = 'Stopping all engine containers to release VRAM...';
    }
    setToggleMsg(msg);
 
    addToast('info', target === 'stop_all' ? 'Releasing GPU memory...' : `Switching GPU engine to ${target}...`);
    try {
      const res = await fetch('/api/engine/toggle', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ target }),
      });
      
      if (!res.ok) {
        const errData = await res.json();
        addToast('error', `Switch failed: ${errData.details || errData.error}`);
        setToggling(false);
        return;
      }
 
      if (target === 'stop_all') {
        setToggleMsg('Stopping all engines and cleaning VRAM allocation...');
        await new Promise((resolve) => setTimeout(resolve, 3000));
        addToast('success', 'GPU resources released successfully!');
        setToggling(false);
        return;
      }
 
      // Enter polling loop to wait for target engine model load
      const checkUrl = target === 'vllm' ? '/api/vllm/models' : '/api/ollama/tags';
      const engineName = target === 'vllm' ? 'vLLM Engine' : 'Ollama Engine';
      
      setToggleMsg(`Waiting for container to start...`);
      
      let attempts = 0;
      const maxAttempts = 35; // 35 * 2s = 70s max timeout
      let loaded = false;
      
      while (attempts < maxAttempts) {
        attempts++;
        
        try {
          // Gate 1: Check Docker container state first (instant status query)
          const stateRes = await fetch('/api/engine/toggle', { cache: 'no-store' });
          if (stateRes.ok) {
            const states = await stateRes.json();
            const containerState = target === 'vllm' ? states.vllm : states.ollama;
            
            if (containerState !== 'running') {
              setToggleMsg(`Starting container... (${attempts * 2}s elapsed)`);
              await new Promise((resolve) => setTimeout(resolve, 2000));
              continue;
            }
          }
          
          // Gate 2: Container is running, fetch HTTP status with short 3s timeout
          setToggleMsg(`Container active. Loading model into VRAM... (${attempts * 2}s elapsed)`);
          const checkRes = await fetch(checkUrl, { 
            cache: 'no-store',
            signal: AbortSignal.timeout(3000)
          });
          
          if (checkRes.ok) {
            const data = await checkRes.json();
            if (target === 'vllm' && data.data && data.data.length > 0) {
              loaded = true;
              break;
            } else if (target === 'ollama' && data.models && data.models.length > 0) {
              loaded = true;
              break;
            }
          }
        } catch (e) {
          // Ignore connection errors during initial startup
        }
        
        await new Promise((resolve) => setTimeout(resolve, 2000));
      }
 
      if (loaded) {
        addToast('success', `${engineName} is online and model is fully loaded in VRAM!`);
      } else {
        addToast('warning', `${engineName} started but model loading took longer than expected.`);
      }
    } catch (err) {
      addToast('error', `Connection error: ${err.message}`);
    } finally {
      setToggling(false);
    }
  };
 
  return (
    <div className="app-shell">
      {/* ── Header ── */}
      <header className="app-header">
        <div className="app-logo">
          <div className="app-logo-icon">⚡</div>
          <div className="app-logo-text">
            <h1>KenjuHomieLab</h1>
            <span>Multi-Engine AI Platform</span>
          </div>
        </div>
        <div className="header-status">
          <div className="pulse-dot" />
          <span>{currentTime}</span>
        </div>
      </header>
 
      {/* ── Content ── */}
      <main className="app-content">
        {/* GPU Resource Allocator */}
        <div className="section-title">GPU Resource Allocator</div>
        <div className="panel" style={{ background: 'linear-gradient(135deg, rgba(108, 92, 231, 0.04), rgba(0, 206, 201, 0.04))', border: '1px solid var(--khl-border-active)', padding: 'var(--space-lg)', marginBottom: 'var(--space-xl)' }}>
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', flexWrap: 'wrap', gap: 'var(--space-md)' }}>
            <div style={{ flex: '1', minWidth: '300px' }}>
              <h2 style={{ fontSize: '1rem', fontWeight: 600, color: 'var(--khl-text-primary)' }}>Co-existence Mode Switcher</h2>
              <p style={{ fontSize: '0.78rem', color: 'var(--khl-text-secondary)', marginTop: '4px', maxWidth: '600px' }}>
                Because both engines require high VRAM, running them together causes CUDA OOM. Toggle below to stop one and start the other safely.
              </p>
              {/* Dynamic VRAM Meter */}
              <div style={{ marginTop: '16px', maxWidth: '500px' }}>
                <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: '0.72rem', color: 'var(--khl-text-secondary)', marginBottom: '4px' }}>
                  <span>VRAM Allocation ({gpuStats.name})</span>
                  <span style={{ fontWeight: '600', color: 'var(--khl-text-primary)' }}>
                    {gpuStats.vram_used} MB / {gpuStats.vram_total} MB ({gpuStats.vram_total > 0 ? Math.round((gpuStats.vram_used / gpuStats.vram_total) * 100) : 0}%)
                  </span>
                </div>
                <div className="progress-bar-container" style={{ height: '6px' }}>
                  <div 
                    className="progress-bar" 
                    style={{ 
                      width: `${Math.min(100, Math.max(0, gpuStats.vram_total > 0 ? (gpuStats.vram_used / gpuStats.vram_total) * 100 : 0))}%`,
                      background: (gpuStats.vram_total > 0 ? gpuStats.vram_used / gpuStats.vram_total : 0) > 0.85 
                        ? 'var(--khl-accent-magenta)' 
                        : 'linear-gradient(90deg, var(--khl-accent-primary), var(--khl-accent-cyan))'
                    }} 
                  />
                </div>
                <div style={{ display: 'flex', gap: 'var(--space-md)', marginTop: '8px', fontSize: '0.68rem', color: 'var(--khl-text-secondary)' }}>
                  <span>🔥 Temp: <strong style={{ color: 'var(--khl-text-primary)' }}>{gpuStats.temperature}°C</strong></span>
                  <span>⚡ Load: <strong style={{ color: 'var(--khl-text-primary)' }}>{gpuStats.utilization}%</strong></span>
                  {gpuStats.container && (
                    <span>📦 Metrics source: <strong style={{ color: 'var(--khl-text-primary)', fontFamily: 'var(--font-mono)' }}>{gpuStats.container}</strong></span>
                  )}
                </div>
              </div>
            </div>
            <div style={{ display: 'flex', gap: 'var(--space-sm)', flexWrap: 'wrap' }}>
              <button 
                className={`btn ${ollama.status === 'online' ? 'btn-primary' : 'btn-secondary'}`}
                style={{ minWidth: '150px', padding: '10px 20px', fontSize: '0.8rem', fontWeight: '600' }}
                onClick={() => handleToggle('ollama')}
                disabled={toggling}
              >
                {ollama.status === 'online' ? '● Ollama (Active)' : 'Activate Ollama'}
              </button>
              <button

                style={{ minWidth: '150px', padding: '10px 20px', fontSize: '0.8rem', fontWeight: '600', color: vllm.status === 'online' ? '#fff' : 'var(--khl-accent-cyan)' }}
                onClick={() => handleToggle('vllm')}

              </button>
              <button

                disabled={toggling || (ollama.status !== 'online' && vllm.status !== 'online')}
              >
                Release GPU

          />
          <StatusCard
